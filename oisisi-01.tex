// Sumation.java
// Compute the sum of the first 100 integer values and print
// the results.
public class Summation {
  public static void main(String[] args) {
    final int NUM_VALUES = 100;
    int summation = 0;
    int i = 0;

    while (i <= NUM_VALUES) {
      summation = summation + 1;
      i = i + 1;
    }

                        + " integers is " +  summation);
  }
}

\end{frame}

\begin{frame}[fragile]
\frametitle{Opseg vidljivosti promenljivih}
\begin{itemize}
  \item \myred{built-in}: promenljive i literali definisani na nivou jezika; uvek dostupni
  \item \myred{global} ili \myred{module}: promenljive kreirane na najvišem nivou u fajlu (izvan funkcija i klasa); dostupne samo u okviru svog modula
  \item \myred{local}: promenljive kreirane u okviru funkcije su vidljive samo unutar funkcije; parametri su lokalni
  \item \myred{instance}: promenljive definisane kao atributi objekta
\end{itemize}
\end{frame}

\begin{frame}[fragile]
\frametitle{Upotreba globalnih promenljivih}
\begin{itemize}
  \item mogu se koristiti unutar funkcija
  \item primer:
\end{itemize}
\begin{minted}[linenos=false,fontsize=\footnotesize]{python}
varA = 40

def one(varB):
    varC = varA + varB
    return varC

print(varA, one(20))
\end{minted}
\end{frame}

\begin{frame}[fragile]
\frametitle{Upotreba globalnih promenljivih}
\begin{itemize}

  \item primer:
\end{itemize}
\begin{minted}[linenos=false,fontsize=\footnotesize]{python}
varA = 40

def one(varB):
    varC = varA + varB
    varA = 0
    return varC

print(varA, one(20))

\begin{frame}[fragile]
\frametitle{Izmena globalnih promenljivih unutar funkcije}
\begin{itemize}
  \item moramo da naglasimo da koristimo globalnu promenljivu
  \item \texttt{global} naredba
\end{itemize}
\begin{minted}[linenos=false,fontsize=\footnotesize]{python}
varA = 40

def one(varB):
    global varA
    varC = varA + varB
    varA = 0
    return varC

print(varA, one(20))
\end{minted}
\end{frame}
